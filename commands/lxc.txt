lxc-checkconfig

lxc-clone

lxc-config

lxc-console

lxc-create

lxc-destroy

lxc-device

lxc-execute

lxc-freeze
# (1) - freeze all the container's processes

lxc-info
# (1) - query information about a container

lxc-ls

lxc-monitor
# (1) - monitor the container state

lxc-snapshot
# (1) - Snapshot an existing container.

lxc-start
# (1) - run an application inside a container.

lxc-start-ephemeral

lxc-stop

lxc-unfreeze

lxc-unshare
